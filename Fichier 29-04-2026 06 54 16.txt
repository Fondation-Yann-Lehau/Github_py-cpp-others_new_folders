```react
import React, { useState } from 'react';
import { 
  Loader2, 
  Settings, 
  Monitor, 
  Package, 
  Layers, 
  ChevronRight,
  Image as ImageIcon,
  Shield,
  Globe,
  Cpu,
  Atom,
  Waves,
  Users
} from 'lucide-react';

const App = () => {
  const [isGenerating, setIsGenerating] = useState(false);
  const [generatedImage, setGeneratedImage] = useState(null);
  const [error, setError] = useState(null);

  // Synchronisation ultime de tous les éléments du contexte
  const assets = [
    { id: 1, icon: <Users className="w-4 h-4"/>, type: 'Profil Utilisateur', name: 'Directeur Stratégie (Tactique)', color: 'bg-orange-500/20 text-orange-400' },
    { id: 2, icon: <Users className="w-4 h-4"/>, type: 'Profil Utilisateur', name: 'Lead Tech (Cyber/Tatouages)', color: 'bg-teal-500/20 text-teal-400' },
    { id: 3, icon: <Cpu className="w-4 h-4"/>, type: 'Inspiration Design', name: 'Armure Cyborg IA (Blanc/Rouge)', color: 'bg-pink-500/20 text-pink-400' },
    { id: 4, icon: <Atom className="w-4 h-4"/>, type: 'Physique Quantique', name: 'Hologrammes d\'Orbitales', color: 'bg-cyan-500/20 text-cyan-400' },
    { id: 5, icon: <Waves className="w-4 h-4"/>, type: 'Physique Quantique', name: 'Équation de Schrödinger', color: 'bg-violet-500/20 text-violet-400' },
    { id: 6, icon: <Shield className="w-4 h-4"/>, type: 'Défense', name: 'GICAT / ScioTeq / GenDyn', color: 'bg-red-500/20 text-red-400' },
    { id: 7, icon: <Globe className="w-4 h-4"/>, type: 'Réseaux & Tech', name: 'Tor / WeChat / NetDragon', color: 'bg-blue-500/20 text-blue-400' }
  ];

  const generateScene = async () => {
    setIsGenerating(true);
    setError(null);
    const apiKey = "";

    // Prompt détaillé qui force l'IA à modéliser un Duo-Pack reprenant les traits des profils et le contexte quantique/défense
    const prompt = `A photorealistic 1/7 scale commercial collector's "Duo Pack" figurine set displayed on a pristine modern white office desk. Figurine 1: A male strategic director with short hair and a neat goatee, wearing an elegant dark tactical defense suit. Figurine 2: A male lead tech cyber-specialist with long dark hair, round glasses, and neck tattoos, wearing sleek white futuristic cybernetic armor with glowing red accents (inspired by advanced robotics). Both figurines are posed dynamically, interacting with glowing cyan and violet holographic projections of quantum mechanics floating between them (atomic orbitals, probability waves, and Schrödinger equation symbols). They stand on a unified transparent acrylic base with subtle defense and cyber-network logos engraved. In the background, a blurred computer monitor and a premium high-end toy packaging box labeled "Quantum Cyber-Defense Taskforce". Cinematic lighting, moody contrast, 8k resolution, ultra-detailed.`;

    const fetchImage = async (retries = 0) => {
      try {
        const response = await fetch(
          `https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`,
          {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              instances: [{ prompt }],
              parameters: { sampleCount: 1 }
            }),
          }
        );

        if (!response.ok) throw new Error('Échec de la génération de l\'image.');

        const data = await response.json();
        const base64Data = data.predictions[0].bytesBase64Encoded;
        setGeneratedImage(`data:image/png;base64,${base64Data}`);
      } catch (err) {
        if (retries < 5) {
          const delay = Math.pow(2, retries) * 1000;
          setTimeout(() => fetchImage(retries + 1), delay);
        } else {
          setError("La connexion au cluster de rendu a échoué. Veuillez vérifier le réseau sécurisé.");
          setIsGenerating(false);
        }
      } finally {
        if (retries === 0 || generatedImage) setIsGenerating(false);
      }
    };

    fetchImage();
  };

  return (
    <div className="min-h-screen bg-[#09090b] text-slate-200 flex flex-col font-sans">
      {/* Header */}
      <header className="bg-[#18181b] border-b border-slate-800 px-6 py-4 flex justify-between items-center sticky top-0 z-10 shadow-md">
        <div className="flex items-center gap-3">
          <div className="bg-gradient-to-br from-cyan-600 to-violet-600 p-2 rounded-lg shadow-[0_0_15px_rgba(139,92,246,0.4)]">
            <Users className="text-white w-5 h-5" />
          </div>
          <div>
            <h1 className="text-xl font-bold text-white tracking-tight">
              Design Orchestrator <span className="text-violet-400">v5.0</span>
            </h1>
            <p className="text-xs text-slate-400 font-medium tracking-wide">Fusion Totale : Profils, Cyber-Défense & Quantique</p>
          </div>
        </div>
        <button className="p-2 hover:bg-slate-800 rounded-full transition-colors">
          <Settings className="w-5 h-5 text-slate-400" />
        </button>
      </header>

      <main className="flex-1 flex flex-col lg:flex-row p-6 gap-6 max-w-7xl mx-auto w-full">
        {/* Sidebar: Assets & Parameters */}
        <div className="w-full lg:w-80 space-y-6 flex flex-col">
          <section className="bg-[#18181b] p-5 rounded-2xl shadow-lg border border-slate-800 flex-1 flex flex-col">
            <h2 className="text-sm font-bold text-slate-400 uppercase tracking-wider mb-4 flex items-center gap-2">
              <Layers className="w-4 h-4 text-violet-500" /> Data Synchronisée
            </h2>
            <div className="grid grid-cols-1 gap-2 flex-1 overflow-y-auto pr-2 scrollbar-thin scrollbar-thumb-slate-700 scrollbar-track-transparent">
              {assets.map((asset) => (
                <div key={asset.id} className={`flex flex-col p-3 rounded-xl ${asset.color} border border-current border-opacity-20 hover:bg-opacity-20 transition-all cursor-default`}>
                  <div className="flex items-center gap-2 mb-1">
                    {asset.icon}
                    <span className="text-[10px] font-bold uppercase tracking-wider opacity-80">{asset.type}</span>
                  </div>
                  <span className="text-sm font-semibold">{asset.name}</span>
                </div>
              ))}
            </div>
          </section>

          <section className="bg-[#18181b] p-5 rounded-2xl shadow-lg border border-slate-800">
            <h2 className="text-sm font-bold text-slate-400 uppercase tracking-wider mb-4">Moteur de Rendu</h2>
            <div className="space-y-4 mb-6">
              <div className="flex justify-between items-center">
                <span className="text-sm text-slate-300 flex items-center gap-2">
                  <Users className="w-4 h-4 text-orange-400" /> Modeleur de Profils
                </span>
                <div className="w-8 h-4 bg-orange-600 rounded-full shadow-[0_0_10px_rgba(249,115,22,0.4)]"></div>
              </div>
              <div className="flex justify-between items-center">
                <span className="text-sm text-slate-300 flex items-center gap-2">
                  <Atom className="w-4 h-4 text-cyan-400" /> Simulation Quantique
                </span>
                <div className="w-8 h-4 bg-cyan-600 rounded-full shadow-[0_0_10px_rgba(8,145,178,0.4)]"></div>
              </div>
            </div>
            <button
              onClick={generateScene}
              disabled={isGenerating}
              className="w-full py-4 bg-gradient-to-r from-violet-600 to-fuchsia-600 text-white rounded-xl font-bold flex items-center justify-center gap-2 hover:from-violet-500 hover:to-fuchsia-500 transition-all shadow-[0_0_20px_rgba(139,92,246,0.4)] active:scale-95 disabled:opacity-50"
            >
              {isGenerating ? (
                <Loader2 className="w-5 h-5 animate-spin" />
              ) : (
                <>Générer le Duo-Pack <ChevronRight className="w-4 h-4" /></>
              )}
            </button>
          </section>
        </div>

        {/* Main Viewport */}
        <div className="flex-1">
          <div className="bg-[#18181b] rounded-3xl shadow-2xl border border-slate-800 overflow-hidden h-full min-h-[600px] flex flex-col relative">
            <div className="p-4 border-b border-slate-800 flex items-center justify-between bg-[#18181b]/80 backdrop-blur-sm z-10">
              <div className="flex items-center gap-2">
                <div className="w-3 h-3 rounded-full bg-red-500"></div>
                <div className="w-3 h-3 rounded-full bg-yellow-500"></div>
                <div className="w-3 h-3 rounded-full bg-green-500"></div>
                <span className="ml-2 text-xs font-mono text-slate-500">render_duo_pack_v5.tiff</span>
              </div>
              <div className="flex gap-2">
                <Badge label="Édition Limitée" color="bg-yellow-500/10 text-yellow-400 border-yellow-500/20" />
                <Badge label="Cyber-Défense Quantique" color="bg-violet-500/10 text-violet-400 border-violet-500/20" />
              </div>
            </div>

            <div className="flex-1 relative flex items-center justify-center bg-[#09090b] p-8 overflow-hidden">
              {/* Grille de fond et halos */}
              <div className="absolute inset-0 bg-[linear-gradient(to_right,#27272a_1px,transparent_1px),linear-gradient(to_bottom,#27272a_1px,transparent_1px)] bg-[size:40px_40px] opacity-20"></div>
              <div className="absolute top-1/2 left-1/3 -translate-x-1/2 -translate-y-1/2 w-[500px] h-[500px] bg-cyan-900/10 rounded-full blur-3xl pointer-events-none"></div>
              <div className="absolute top-1/2 right-1/3 translate-x-1/2 -translate-y-1/2 w-[500px] h-[500px] bg-violet-900/10 rounded-full blur-3xl pointer-events-none"></div>

              {!generatedImage && !isGenerating && (
                <div className="text-center space-y-4 relative z-10">
                  <div className="bg-slate-800/40 p-6 rounded-full shadow-[0_0_30px_rgba(139,92,246,0.2)] border border-slate-700 inline-block mb-4 backdrop-blur-md">
                    <Users className="w-12 h-12 text-violet-400" />
                  </div>
                  <h3 className="text-xl font-bold text-white">Prêt pour la Fusion Ultime</h3>
                  <p className="text-slate-400 max-w-md mx-auto text-sm leading-relaxed">
                    Les caractéristiques physiques des profils utilisateurs ont été appliquées aux modèles 3D. Le personnage stratégique et le cyborg tactique (Lead Tech) vont être générés avec l'intégration des éléments de Défense et des hologrammes Quantiques.
                  </p>
                </div>
              )}

              {isGenerating && (
                <div className="flex flex-col items-center relative z-10">
                  <div className="relative">
                    <div className="w-24 h-24 border-4 border-slate-800 border-t-cyan-400 border-b-violet-400 rounded-full animate-spin"></div>
                    <Atom className="w-8 h-8 text-cyan-400 absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 animate-pulse" />
                  </div>
                  <p className="mt-6 font-mono text-violet-400 animate-pulse tracking-widest uppercase text-xs">
                    Synthèse des Modèles 3D en cours...
                  </p>
                </div>
              )}

              {generatedImage && (
                <div className="relative group w-full h-full flex items-center justify-center z-10">
                  <img 
                    src={generatedImage} 
                    alt="Figurine Duo Pack - Profils Utilisateurs" 
                    className="max-w-full max-h-full rounded-2xl shadow-[0_0_60px_rgba(139,92,246,0.15)] block border border-slate-700 object-contain relative z-10"
                  />
                  <div className="absolute inset-0 bg-gradient-to-t from-slate-900 via-transparent to-transparent opacity-0 group-hover:opacity-100 transition-opacity duration-300 rounded-2xl pointer-events-none"></div>
                </div>
              )}

              {error && (
                <div className="absolute bottom-8 left-1/2 -translate-x-1/2 px-6 py-3 bg-red-900/90 border border-red-500 text-red-100 rounded-full shadow-lg text-sm font-medium z-20 backdrop-blur-sm flex items-center gap-2">
                  <span className="w-2 h-2 rounded-full bg-red-400 animate-pulse"></span>
                  {error}
                </div>
              )}
            </div>
          </div>
        </div>
      </main>
    </div>
  );
};

// Composant utilitaire pour les badges
const Badge = ({ label, color }) => (
  <span className={`text-[10px] uppercase tracking-wider font-bold px-2 py-1 rounded border ${color} shadow-sm backdrop-blur-sm`}>
    {label}
  </span>
);

export default App;


```
