```react
import React, { useState } from 'react';
import { 
  Loader2, 
  Settings, 
  Monitor, 
  Package, 
  Layers, 
  ChevronRight,
  Image as ImageIcon,
  Shield,
  Globe,
  Cpu,
  Atom,
  Waves,
  Zap
} from 'lucide-react';

const App = () => {
  const [isGenerating, setIsGenerating] = useState(false);
  const [generatedImage, setGeneratedImage] = useState(null);
  const [error, setError] = useState(null);

  // Synchronisation exhaustive : Cyber-Défense, Réseaux Globaux, Profils et Mécanique Quantique
  const assets = [
    { id: 1, icon: <Globe className="w-4 h-4"/>, type: 'Tech/Gaming', name: 'NetDragon', color: 'bg-blue-100 text-blue-600' },
    { id: 2, icon: <Shield className="w-4 h-4"/>, type: 'Privacy', name: 'Tor Project', color: 'bg-purple-100 text-purple-600' },
    { id: 3, icon: <Globe className="w-4 h-4"/>, type: 'Social', name: 'WeChat', color: 'bg-green-100 text-green-600' },
    { id: 4, icon: <Shield className="w-4 h-4"/>, type: 'Défense', name: 'GICAT', color: 'bg-red-100 text-red-600' },
    { id: 5, icon: <Monitor className="w-4 h-4"/>, type: 'Aviation Tech', name: 'ScioTeq', color: 'bg-slate-200 text-slate-800' },
    { id: 6, icon: <Shield className="w-4 h-4"/>, type: 'Aérospatial', name: 'General Dynamics', color: 'bg-blue-200 text-blue-800' },
    { id: 7, icon: <Globe className="w-4 h-4"/>, type: 'Think Tank', name: 'COHA', color: 'bg-indigo-100 text-indigo-600' },
    { id: 8, icon: <ImageIcon className="w-4 h-4"/>, type: 'Profil Utilisateur', name: 'Directeur Stratégie', color: 'bg-orange-100 text-orange-600' },
    { id: 9, icon: <ImageIcon className="w-4 h-4"/>, type: 'Profil Utilisateur', name: 'Lead Tech Cyber', color: 'bg-teal-100 text-teal-600' },
    { id: 10, icon: <Cpu className="w-4 h-4"/>, type: 'Inspiration', name: 'Cyborg IA', color: 'bg-pink-100 text-pink-600' },
    // Nouveaux assets issus de la Mécanique Quantique
    { id: 11, icon: <Atom className="w-4 h-4"/>, type: 'Physique Quantique', name: 'Équation de Schrödinger', color: 'bg-cyan-100 text-cyan-700' },
    { id: 12, icon: <Waves className="w-4 h-4"/>, type: 'Physique Quantique', name: 'Dualité Onde-Particule', color: 'bg-violet-100 text-violet-700' },
    { id: 13, icon: <Zap className="w-4 h-4"/>, type: 'Modèle Atomique', name: 'Densité de Probabilité', color: 'bg-yellow-100 text-yellow-700' }
  ];

  const generateScene = async () => {
    setIsGenerating(true);
    setError(null);
    const apiKey = "";

    // Prompt mis à jour pour intégrer les éléments visuels de la mécanique quantique
    const prompt = `A highly realistic professional office environment. On a clean modern white desk sits a 1/7 scale commercial figurine of a futuristic sleek humanoid AI cyborg with smooth white panels and subtle glowing neon accents. Surrounding the cyborg are subtle, luminous holographic projections of quantum mechanics phenomena: glowing atomic orbitals, probability density waves, and the Schrödinger equation mathematical symbols. The figurine stands on a perfectly round transparent acrylic base. In the background, a large computer monitor displays a technical 3D modeling interface blending cyber-defense wireframes with quantum wavefunction graphs. Next to the monitor, a premium high-end toy packaging box features sophisticated global tech, secure network, and aerospace defense corporate branding logos. The lighting is cinematic, moody and soft, with a shallow depth of field focusing on the quantum-enhanced figurine. 8k resolution, photorealistic.`;

    const fetchImage = async (retries = 0) => {
      try {
        const response = await fetch(
          `https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`,
          {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              instances: [{ prompt }],
              parameters: { sampleCount: 1 }
            }),
          }
        );

        if (!response.ok) throw new Error('Échec de la génération');

        const data = await response.json();
        const base64Data = data.predictions[0].bytesBase64Encoded;
        setGeneratedImage(`data:image/png;base64,${base64Data}`);
      } catch (err) {
        if (retries < 5) {
          const delay = Math.pow(2, retries) * 1000;
          setTimeout(() => fetchImage(retries + 1), delay);
        } else {
          setError("La connexion au studio de rendu a échoué. Veuillez réessayer.");
          setIsGenerating(false);
        }
      } finally {
        if (retries === 0 || generatedImage) setIsGenerating(false);
      }
    };

    fetchImage();
  };

  return (
    <div className="min-h-screen bg-[#0f172a] text-slate-200 flex flex-col font-sans">
      {/* Header */}
      <header className="bg-slate-900 border-b border-slate-800 px-6 py-4 flex justify-between items-center sticky top-0 z-10 shadow-md">
        <div className="flex items-center gap-3">
          <div className="bg-cyan-600 p-2 rounded-lg shadow-[0_0_15px_rgba(8,145,178,0.5)]">
            <Atom className="text-white w-5 h-5" />
          </div>
          <div>
            <h1 className="text-xl font-bold text-white tracking-tight">
              Design Orchestrator <span className="text-cyan-400">v4.0</span>
            </h1>
            <p className="text-xs text-slate-400 font-medium tracking-wide">Cyber-Défense & Mécanique Quantique</p>
          </div>
        </div>
        <button className="p-2 hover:bg-slate-800 rounded-full transition-colors">
          <Settings className="w-5 h-5 text-slate-400" />
        </button>
      </header>

      <main className="flex-1 flex flex-col lg:flex-row p-6 gap-6 max-w-7xl mx-auto w-full">
        {/* Sidebar: Assets & Parameters */}
        <div className="w-full lg:w-80 space-y-6 flex flex-col">
          <section className="bg-slate-900 p-5 rounded-2xl shadow-lg border border-slate-800 flex-1 flex flex-col">
            <h2 className="text-sm font-bold text-slate-400 uppercase tracking-wider mb-4 flex items-center gap-2">
              <Layers className="w-4 h-4 text-cyan-500" /> Base de données
            </h2>
            <div className="grid grid-cols-1 gap-2 flex-1 overflow-y-auto pr-2 scrollbar-thin scrollbar-thumb-slate-700 scrollbar-track-transparent">
              {assets.map((asset) => (
                <div key={asset.id} className={`flex items-center justify-between p-2.5 rounded-xl ${asset.color} bg-opacity-10 border border-current border-opacity-20 hover:bg-opacity-20 transition-all cursor-default`}>
                  <div className="flex items-center gap-2">
                    {asset.icon}
                    <span className="text-xs font-bold opacity-80">{asset.type}</span>
                  </div>
                  <span className="text-sm font-semibold">{asset.name}</span>
                </div>
              ))}
            </div>
          </section>

          <section className="bg-slate-900 p-5 rounded-2xl shadow-lg border border-slate-800">
            <h2 className="text-sm font-bold text-slate-400 uppercase tracking-wider mb-4">Configuration Rendu</h2>
            <div className="space-y-4">
              <div className="flex justify-between items-center">
                <span className="text-sm text-slate-300 flex items-center gap-2">
                  <Cpu className="w-4 h-4 text-cyan-400" /> Modélisation Cyborg IA
                </span>
                <div className="w-8 h-4 bg-cyan-600 rounded-full shadow-[0_0_10px_rgba(8,145,178,0.5)]"></div>
              </div>
              <div className="flex justify-between items-center">
                <span className="text-sm text-slate-300 flex items-center gap-2">
                  <Atom className="w-4 h-4 text-violet-400" /> Hologrammes Quantiques
                </span>
                <div className="w-8 h-4 bg-violet-600 rounded-full shadow-[0_0_10px_rgba(124,58,237,0.5)]"></div>
              </div>
              <div className="flex justify-between items-center">
                <span className="text-sm text-slate-300 flex items-center gap-2">
                  <Package className="w-4 h-4 text-slate-400" /> Packaging Corporate
                </span>
                <div className="w-8 h-4 bg-slate-600 rounded-full"></div>
              </div>
            </div>
            <button
              onClick={generateScene}
              disabled={isGenerating}
              className="w-full mt-6 py-4 bg-gradient-to-r from-cyan-600 to-blue-600 text-white rounded-xl font-bold flex items-center justify-center gap-2 hover:from-cyan-500 hover:to-blue-500 transition-all shadow-[0_0_20px_rgba(8,145,178,0.3)] active:scale-95 disabled:opacity-50"
            >
              {isGenerating ? (
                <Loader2 className="w-5 h-5 animate-spin" />
              ) : (
                <>Générer l'Avant-Projet Quantique <ChevronRight className="w-4 h-4" /></>
              )}
            </button>
          </section>
        </div>

        {/* Main Viewport */}
        <div className="flex-1">
          <div className="bg-slate-900 rounded-3xl shadow-2xl border border-slate-800 overflow-hidden h-full min-h-[600px] flex flex-col relative">
            <div className="p-4 border-b border-slate-800 flex items-center justify-between bg-slate-900/50 backdrop-blur-sm z-10">
              <div className="flex items-center gap-2">
                <div className="w-3 h-3 rounded-full bg-red-500"></div>
                <div className="w-3 h-3 rounded-full bg-yellow-500"></div>
                <div className="w-3 h-3 rounded-full bg-green-500"></div>
                <span className="ml-2 text-xs font-mono text-slate-500">render_quantum_defense.tiff</span>
              </div>
              <div className="flex gap-2">
                <Badge label="Confidentiel Défense" color="bg-red-500/10 text-red-400 border-red-500/20" />
                <Badge label="Réseau Sécurisé" color="bg-purple-500/10 text-purple-400 border-purple-500/20" />
                <Badge label="Superposition Quantique" color="bg-cyan-500/10 text-cyan-400 border-cyan-500/20" />
              </div>
            </div>

            <div className="flex-1 relative flex items-center justify-center bg-[#020617] p-8 overflow-hidden">
              {/* Grille de fond style cyber/quantique */}
              <div className="absolute inset-0 bg-[linear-gradient(to_right,#1e293b_1px,transparent_1px),linear-gradient(to_bottom,#1e293b_1px,transparent_1px)] bg-[size:32px_32px] opacity-30"></div>
              
              {/* Effet d'onde de fond (simulation orbitale) */}
              <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-[800px] h-[800px] bg-cyan-900/10 rounded-full blur-3xl pointer-events-none"></div>
              <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-[400px] h-[400px] bg-violet-900/10 rounded-full blur-3xl pointer-events-none"></div>

              {!generatedImage && !isGenerating && (
                <div className="text-center space-y-4 relative z-10">
                  <div className="bg-slate-800/50 p-6 rounded-full shadow-[0_0_30px_rgba(8,145,178,0.2)] border border-slate-700 inline-block mb-4 backdrop-blur-sm">
                    <Atom className="w-12 h-12 text-cyan-400 animate-[spin_4s_linear_infinite]" />
                  </div>
                  <h3 className="text-xl font-bold text-white">Analyse Quantique & Stratégique</h3>
                  <p className="text-slate-400 max-w-md mx-auto text-sm leading-relaxed">
                    Les données de Défense (GICAT, ScioTeq, General Dynamics), les réseaux (Tor, WeChat) et les principes de la mécanique quantique (Schrödinger, Densité de probabilité) ont été synchronisés avec l'inspiration IA.
                  </p>
                </div>
              )}

              {isGenerating && (
                <div className="flex flex-col items-center relative z-10">
                  <div className="relative">
                    <div className="w-24 h-24 border-4 border-slate-700 border-t-cyan-400 border-r-violet-400 rounded-full animate-spin"></div>
                    <Atom className="w-8 h-8 text-cyan-400 absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2" />
                  </div>
                  <p className="mt-6 font-mono text-cyan-400 animate-pulse tracking-widest uppercase text-xs">
                    Calcul de la fonction d'onde... $\hat{"{H}"}\Psi = E\Psi$
                  </p>
                </div>
              )}

              {generatedImage && (
                <div className="relative group w-full h-full flex items-center justify-center z-10">
                  <img 
                    src={generatedImage} 
                    alt="Figurine Cyborg Collector - Modèle Quantique" 
                    className="max-w-full max-h-full rounded-2xl shadow-[0_0_50px_rgba(8,145,178,0.15)] block border border-slate-700 object-contain relative z-10"
                  />
                  <div className="absolute inset-0 bg-gradient-to-t from-slate-900 via-transparent to-transparent opacity-0 group-hover:opacity-100 transition-opacity duration-300 rounded-2xl pointer-events-none"></div>
                </div>
              )}

              {error && (
                <div className="absolute bottom-8 left-1/2 -translate-x-1/2 px-6 py-3 bg-red-900/90 border border-red-500 text-red-100 rounded-full shadow-lg text-sm font-medium z-20 backdrop-blur-sm flex items-center gap-2">
                  <span className="w-2 h-2 rounded-full bg-red-400 animate-pulse"></span>
                  {error}
                </div>
              )}
            </div>
          </div>
        </div>
      </main>
    </div>
  );
};

// Composant utilitaire pour les badges
const Badge = ({ label, color }) => (
  <span className={`text-[10px] uppercase tracking-wider font-bold px-2 py-1 rounded border ${color} shadow-sm`}>
    {label}
  </span>
);

export default App;
```
