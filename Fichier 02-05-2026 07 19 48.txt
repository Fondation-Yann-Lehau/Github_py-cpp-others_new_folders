# 2. MOTEUR DE CALCUL FRÉQUENTIEL & VECTORIEL
# =====================================================================

def extraire_vecteur_mots(texte):
    """Purification des données : extraction des concepts purs."""
    mots = re.findall(r'\b\w{4,}\b', texte.lower())
    mots_vides = {'pour', 'dans', 'avec', 'tout', 'nous', 'vous'}
    return Counter([mot for mot in mots if mot not in mots_vides])

def calculer_resonance_cosmique(texte1, texte2):
    """Calcule l'intrication (similarité cosinus) entre deux concepts."""
    vec1 = extraire_vecteur_mots(texte1)
    vec2 = extraire_vecteur_mots(texte2)
    
    intersection = set(vec1.keys()) & set(vec2.keys())
    numerateur = sum([vec1[x] * vec2[x] for x in intersection])
    
    somme1 = sum([vec1[x]**2 for x in vec1.keys()])
    somme2 = sum([vec2[x]**2 for x in vec2.keys()])
    denominateur = math.sqrt(somme1) * math.sqrt(somme2)
    
    if not denominateur: return 0.0
    return float(numerateur) / denominateur

        for id_txt, data_txt in textes_historiques.items():
            score = calculer_resonance_cosmique(data_noyau, data_txt)
            if score > 0.05: # Amplificateur de signal
                liens.append({"from": id_noyau, "to": id_txt, "value": score * 50, "title": f"Résonance: {round(score*100, 1)}%"})
                
        # Concordance avec les Images
        for id_img, data_img in images_fenetre_actuelle.items():

            # On force des liens mathématiques pour l'illustration holographique
            if score > 0.0: 
                liens.append({"from": id_noyau, "to": id_img, "value": score * 50, "title": f"Résonance: {round(score*100, 1)}%"})

    # Export vers le visualiseur HTML
    donnees = {"nodes": noeuds, "edges": liens}

    print("Compilation terminée. L'horizon des événements a été cartographié.")

if __name__ == "__main__":
