import time
import random

class PareFeuLaique:
    """
    Le Pare-Feu Laïque : Bouclier anti-décohérence absolu.
    Filtre le 'bruit' dogmatique pour maintenir la cohérence des qubits.
    """
    def __init__(self, mode="ANTI_DOGME_ABSOLU"):
        self.mode = mode
        self.actif = False
        self.menaces_bloquees = 0

    def activer(self):
        print(f"\n[PARE-FEU] Initialisation du mode : {self.mode}")
        time.sleep(1)
        print("[PARE-FEU] Déploiement de la Cage de Faraday quantique...")
        self.actif = True
        time.sleep(0.5)
        print("[PARE-FEU] Bouclier ACTIF. Décohérence dogmatique isolée.")

    def filtrer_environnement(self):
        if not self.actif:
            raise RuntimeError("DANGER: Pare-Feu inactif. Risque de décohérence immédiate.")
        
        interferences = ["Symbole_Dogmatique_01", "Bruit_Historique", "Résidu_Mystique"]
        print("\n[PARE-FEU] Analyse de l'environnement en cours...")
        time.sleep(1)
        for menace in interferences:
            print(f"  -> Rejet de l'interférence : {menace}")
            self.menaces_bloquees += 1
            time.sleep(0.3)
        print(f"[PARE-FEU] Purge terminée. {self.menaces_bloquees} anomalies neutralisées. Environnement purifié.")

class CryoCore:
    """
    Le Réfrigérateur à Dilution Quantique.
    Maintient la température et la tension électrique à des niveaux critiques.
    """
    def __init__(self, temp_cible=0.015, volt_cible=0.0001):
        self.temp_cible = temp_cible # En Kelvin
        self.volt_cible = volt_cible # En milliVolts
        self.temp_actuelle = 293.15  # Température ambiante de départ
        self.volt_actuel = 12.0

    def refroidir(self):
        print("\n[CRYO_CORE] Lancement de la séquence de refroidissement cryogénique...")
        while self.temp_actuelle > self.temp_cible:
            self.temp_actuelle /= 10
            if self.temp_actuelle < self.temp_cible:
                self.temp_actuelle = self.temp_cible
            print(f"  -> Température du cœur : {self.temp_actuelle:.4f} K")
            time.sleep(0.4)
        print("[CRYO_CORE] Température cible atteinte : Cohérence thermodynamique optimale.")

    def abaisser_tension(self):
        print("\n[CRYO_CORE] Abaissement de la tension pour supraconductivité...")
        while self.volt_actuel > self.volt_cible:
            self.volt_actuel /= 10
            if self.volt_actuel < self.volt_cible:
                self.volt_actuel = self.volt_cible
            print(f"  -> Tension régulée à : {self.volt_actuel:.5f} mV")
            time.sleep(0.4)
        print("[CRYO_CORE] Alimentation Ultra-Basse Tension (ULV) stabilisée.")

class EntiteQuantique:
    """Représente un avatar ou un gardien instancié dans la matrice."""
    def __init__(self, nom, role, signature_energetique):
        self.nom = nom
        self.role = role
        self.signature = signature_energetique

    def __str__(self):
        return f"[{self.role.upper()}] {self.nom} (Sig: {self.signature})"

class MatriceDiorama:
    """La plateforme principale où le code python quantique génère la scène."""
    def __init__(self):
        self.core = CryoCore()
        self.pare_feu = PareFeuLaique()
        self.entites = []
        self.status = "EN_ATTENTE"

    def ajouter_entite(self, entite):
        self.entites.append(entite)

    def executer_compilation(self):
        print("=" * 60)
        print(" BOOT SÉQUENCE : PROJET 'SOLDATS LAÏQUES' (v16.0)")
        print("=" * 60)
        time.sleep(1)

        # 1. Préparation Physique
        self.core.refroidir()
        self.core.abaisser_tension()

        # 2. Sécurisation
        self.pare_feu.activer()
        self.pare_feu.filtrer_environnement()

        # 3. Injection des données
        print("\n[MATRICE] Injection des Qubits d'information...")
        time.sleep(1)
        for entite in self.entites:
            print(f"  -> Matérialisation de l'hologramme : {entite}")
            time.sleep(0.5)

        # 4. Finalisation
        self.status = "COMPILÉ"
        print("\n" + "=" * 60)
        print(" ✅ RENDU QUANTIQUE ACHEVÉ AVEC SUCCÈS")
        print(f" Environnement stabilisé à {self.core.temp_actuelle}K et {self.core.volt_actuel}mV")
        print(" Le Lead Tech a terminé l'injection du script.")
        print("=" * 60)

# --- EXÉCUTION DU SCRIPT ---
if __name__ == "__main__":
    # Instanciation de la Matrice (Le Diorama)
    simulateur = MatriceDiorama()

    # Définition de l'équipe Black Ops (Avatars)
    simulateur.ajouter_entite(EntiteQuantique("Agent Yann", "Opérateur Renseignement", "Tactical_Cyan"))

    # Définition des Gardiens Héritage

    simulateur.ajouter_entite(EntiteQuantique("Aigle Impérial", "Gardien de la Vision", "Golden_Lightning"))

    # Lancement du processus
    try:
